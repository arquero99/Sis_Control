trans_primer_orden(C_inf,t_63,T):=
block([a,b,k],
    k=t_63/T,
    a=%e**(log(1-0.63)/(k+1)),
    b=(1-a)*C_inf,
return ((b*z)/(z-a))
);


trans_segundo_orden(tp,ts,C_inf,T):=
block([α,σ,k,P1,P2,denom],
    α=(%pi/tp)*T,
    σ=(%pi/ts)*T,
    P1=%e**(-σ)*(cos(α)+%i*sin(α)),
    P2=conjugate(P1),
    denom=expand((z-P1)*(z-P2)),
    k=C_inf*subst(1,z,denom),
return(k/((z-P1)*(z-P2)))
);
